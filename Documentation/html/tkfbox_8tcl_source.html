1457 # tk::dialog::file::ResolveFile --
1458 #
1459 # Interpret the user's text input in a file selection dialog.
1460 # Performs:
1461 #
1462 # (1) ~ substitution
1463 # (2) resolve all instances of . and ..
1464 # (3) check for non-existent files/directories
1465 # (4) check for chdir permissions
1466 # (5) conversion of environment variable references to their
1467 # contents (once only)
1468 #
1469 # Arguments:
1470 # context: the current directory you are in
1471 # text: the text entered by the user
1472 # defaultext: the default extension to add to files with no extension
1473 # expandEnv: whether to expand environment variables (yes by default)
1474 #
1475 # Return vaue:
1476 # [list $flag $directory $file]
1477 #
1478 # flag = OK : valid input
1479 # = PATTERN : valid directory/pattern
1480 # = PATH : the directory does not exist
1481 # = FILE : the directory exists by the file doesn't
1482 # exist
1483 # = CHDIR : Cannot change to the directory
1484 # = ERROR : Invalid entry
1485 #
1486 # directory : valid only if flag = OK or PATTERN or FILE
1487 # file : valid only if flag = OK or PATTERN
1488 #
1489 # directory may not be the same as context, because text may contain
1490 # a subdirectory name
1491 #

1494

1498  # to do this for directories, otherwise typing a dirname in the box
1499  # will give back "dirname.extension" instead of trying to change dir.
1500  if {
1501  ![file isdirectory $path] && ([file ext $path] eq "") &&
1502  ![string match {$*} [file tail $path]]
1503  } then {
1504  set path "$path$defaultext"
1505  }
1506 
1507  if {[catch {file exists $path}]} {
1508  # This "if" block can be safely removed if the following code
1509  # stop generating errors.
1510  #
1511  # file exists ~nonsuchuser
1512  #
1513  return [list ERROR $path ""]
1514  }
1515 
1516  if {[file exists $path]} {
1517  if {[file isdirectory $path]} {
1518  if {[catch {cd $path}]} {
1519  return [list CHDIR $path ""]
1520  }
1521  set directory [pwd]
1522  set file ""
1523  set flag OK
1524  cd $appPWD
1525  } else {
1526  if {[catch {cd [file dirname $path]}]} {
1527  return [list CHDIR [file dirname $path] ""]
1528  }
1529  set directory [pwd]
1530  set file [file tail $path]
1531  set flag OK
1532  cd $appPWD
1533  }
1534  } else {
1535  set dirname [file dirname $path]
1536  if {[file exists $dirname]} {
1537  if {[catch {cd $dirname}]} {
1538  return [list CHDIR $dirname ""]
1539  }
1540  set directory [pwd]
1541  cd $appPWD
1542  set file [file tail $path]
1543  # It's nothing else, so check to see if it is an env-reference
1544  if {$expandEnv && [string match {$*} $file]} {
1545  set var [string range $file 1 end]
1546  if {[info exist ::env($var)]} {
1547  return [ResolveFile $context $::env($var) $defaultext 0]
1548  }
1549  }
1550  if {[regexp {[*?]} $file]} {
1551  set flag PATTERN
1552  } else {
1553  set flag FILE
1554  }
1555  } else {
1556  set directory $dirname
1557  set file [file tail $path]
1558  set flag PATH
1559  # It's nothing else, so check to see if it is an env-reference
1560  if {$expandEnv && [string match {$*} $file]} {
1561  set var [string range $file 1 end]
1562  if {[info exist ::env($var)]} {
1563  return [ResolveFile $context $::env($var) $defaultext 0]
1564  }
1565  }
1566  }
1567  }
1568
