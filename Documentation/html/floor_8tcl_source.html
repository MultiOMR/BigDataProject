14 # floorDisplay --
15 # Recreate the floorplan display in the canvas given by "w". The
16 # floor given by "active" is displayed on top with its office structure
17 # visible.
18 #
19 # Arguments:
20 # w - Name of the canvas window.
21 # active - Number of active floor (1, 2, or 3).
22 
23 proc floorDisplay {w active} {
24  global floorLabels floorItems colors activeFloor
25 
26  if {$activeFloor == $active} {
27  return
28  }
29 
30  $w delete all
31  set activeFloor $active
32 
33  # First go through the three floors, displaying the backgrounds for
34  # each floor.
35 
36  bg1 $w $colors(bg1) $colors(outline1)
37  bg2 $w $colors(bg2) $colors(outline2)
38  bg3 $w $colors(bg3) $colors(outline3)
39 
40  # Raise the background for the active floor so that it's on top.
41 
42  $w raise floor$active
43 
44  # Create a dummy item just to mark this point in the display list,
45  # so we can insert highlights here.
46 
47  $w create rect 0 100 1 101 -fill {} -outline {} -tags marker
48 
49  # Add the walls and labels for the active floor, along with
50  # transparent polygons that define the rooms on the floor.
51  # Make sure that the room polygons are on top.
52 
53  catch {unset floorLabels}
54  catch {unset floorItems}
55  fg$active $w $colors(offices)
56  $w raise room
57 
58  # Offset the floors diagonally from each other.
59 
60  $w move floor1 2c 2c
61  $w move floor2 1c 1c
62 
63  # Create items for the room entry and its label.
64 
65  $w create window 600 100 -anchor w -window $w.entry
66  $w create text 600 100 -anchor e -text "Room: "
67  $w config -scrollregion [$w bbox all]
68 }
