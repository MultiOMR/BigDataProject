78 proc scrollEnter canvas {
79  global oldFill
80  set id [$canvas find withtag current]
81  if {[lsearch [$canvas gettags current] text] >= 0} {
82  set id [expr {$id-1}]

84  set oldFill [lindex [$canvas itemconfig $id -fill] 4]
