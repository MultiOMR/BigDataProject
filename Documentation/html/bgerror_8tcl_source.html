80 # ::tk::dialog::error::bgerror --
81 # This is the default version of bgerror.
82 # It tries to execute tkerror, if that fails it posts a dialog box containing
83 # the error message and gives the user a chance to ask to see a stack
84 # trace.
85 # Arguments:
86 # err - The error message.
87 
88 proc ::tk::dialog::error::bgerror err {
89  global errorInfo tcl_platform
90  variable button
91 
92  set info $errorInfo
93 
94  set ret [catch {::tkerror $err} msg];
95  if {$ret != 1} {return -code $ret $msg}
96 
97  # Ok the application's tkerror either failed or was not found
98  # we use the default dialog then :
99  set windowingsystem [tk windowingsystem]
100  if {$windowingsystem eq "aqua"} {
101  set ok [mc Ok]
102  } else {
103  set ok [mc OK]
104  }
105 
106  # Truncate the message if it is too wide (>maxLine characters) or
107  # too tall (>4 lines). Truncation occurs at the first point at
108  # which one of those conditions is met.
109  set displayedErr ""
110  set lines 0
111  set maxLine 45
112  foreach line [split $err \n] {
113  if { [string length $line] > $maxLine } {
114  append displayedErr "[string range $line 0 [expr {$maxLine-3}]]..."
115  break
116  }
117  if { $lines > 4 } {
118  append displayedErr "..."
119  break
120  } else {
121  append displayedErr "${line}\n"
122  }
123  incr lines
124  }
125 
126  set title [mc "Application Error"]
127  set text [mc "Error: %1\$s" $displayedErr]
128  set buttons [list ok $ok dismiss [mc "Skip Messages"] \
129  function [mc "Details >>"]]
130 
131  # 1. Create the top-level window and divide it into top
132  # and bottom parts.
133 
134  set dlg .bgerrorDialog
135  destroy $dlg
136  toplevel $dlg -class ErrorDialog
137  wm withdraw $dlg
138  wm title $dlg $title
139  wm iconname $dlg ErrorDialog
140  wm protocol $dlg WM_DELETE_WINDOW { }
141 
142  if {$windowingsystem eq "aqua"} {
143  ::tk::unsupported::MacWindowStyle style $dlg moveableAlert {}
144  }
145 
146  frame $dlg.bot
147  frame $dlg.top
148  if {$windowingsystem eq "x11"} {
149  $dlg.bot configure -relief raised -bd 1
150  $dlg.top configure -relief raised -bd 1
151  }
152  pack $dlg.bot -side bottom -fill both
153  pack $dlg.top -side top -fill both -expand 1
154 
155  set W [frame $dlg.top.info]
156  text $W.text -setgrid true -height 10 -wrap char \
157  -yscrollcommand [list $W.scroll set]
158  if {$windowingsystem ne "aqua"} {
159  $W.text configure -width 40
160  }
161 
162  scrollbar $W.scroll -command [list $W.text yview]
163  pack $W.scroll -side right -fill y
164  pack $W.text -side left -expand yes -fill both
165  $W.text insert 0.0 "$err\n$info"
166  $W.text mark set insert 0.0
167  bind $W.text <ButtonPress-1> { focus %W }
168  $W.text configure -state disabled
169 
170  # 2. Fill the top part with bitmap and message
171 
172  # Max-width of message is the width of the screen...
173  set wrapwidth [winfo screenwidth $dlg]
174  # ...minus the width of the icon, padding and a fudge factor for
175  # the window manager decorations and aesthetics.
176  set wrapwidth [expr {$wrapwidth-60-[winfo pixels $dlg 9m]}]
177  label $dlg.msg -justify left -text $text -wraplength $wrapwidth
178  if {$windowingsystem eq "aqua"} {
179  # On the Macintosh, use the stop bitmap
180  label $dlg.bitmap -bitmap stop
181  } else {
182  # On other platforms, make the error icon
183  canvas $dlg.bitmap -width 32 -height 32 -highlightthickness 0
184  $dlg.bitmap create oval 0 0 31 31 -fill red -outline black
185  $dlg.bitmap create line 9 9 23 23 -fill white -width 4
186  $dlg.bitmap create line 9 23 23 9 -fill white -width 4
187  }
188  grid $dlg.bitmap $dlg.msg -in $dlg.top -row 0 -padx 3m -pady 3m
189  grid configure $dlg.msg -sticky nsw -padx {0 3m}
190  grid rowconfigure $dlg.top 1 -weight 1
191  grid columnconfigure $dlg.top 1 -weight 1
192 
193  # 3. Create a row of buttons at the bottom of the dialog.
194 
195  set i 0
196  foreach {name caption} $buttons {
197  button $dlg.$name -text $caption -default normal \
198  -command [namespace code [list set button $i]]
199  grid $dlg.$name -in $dlg.bot -column $i -row 0 -sticky ew -padx 10
200  grid columnconfigure $dlg.bot $i -weight 1
201  # We boost the size of some Mac buttons for l&f
202  if {$windowingsystem eq "aqua"} {
203  if {($name eq "ok") || ($name eq "dismiss")} {
204  grid columnconfigure $dlg.bot $i -minsize 90
205  }
206  grid configure $dlg.$name -pady 7
207  }
208  incr i
209  }
210  # The "OK" button is the default for this dialog.
211  $dlg.ok configure -default active
212 
213  bind $dlg <Return> [namespace code Return]
214  bind $dlg <Destroy> [namespace code [list Destroy %W]]
215  $dlg.function configure -command [namespace code Details]
216 
217  # 6. Place the window (centered in the display) and deiconify it.
218 
219  ::tk::PlaceWindow $dlg
220 
221  # 7. Ensure that we are topmost.
222 
223  raise $dlg
224  if {$tcl_platform(platform) eq "windows"} {
225  # Place it topmost if we aren't at the top of the stacking
226  # order to ensure that it's seen
227  if {[lindex [wm stackorder .] end] ne "$dlg"} {
228  wm attributes $dlg -topmost 1
229  }
230  }
231 
232  # 8. Set a grab and claim the focus too.
233 
234  ::tk::SetFocusGrab $dlg $dlg.ok
235 
236  # 9. Wait for the user to respond, then restore the focus and
237  # return the index of the selected button. Restore the focus
238  # before deleting the window, since otherwise the window manager
239  # may take the focus away so we can't redirect it. Finally,
240  # restore any grab that was in effect.
241 
242  vwait [namespace which -variable button]
243  set copy $button; # Save a copy...
244 
245  ::tk::RestoreFocusGrab $dlg $dlg.ok destroy
246 
247  if {$copy == 1} {
248  return -code break
249  }
250 }
